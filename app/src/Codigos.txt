

  import { useState } from 'react'
import './App.css'



function App() {
  const [count, setCount] = useState<number>(0)


  const addCount = () => {  
    setCount(prev => prev + 1) 
  }

  const removeCount = () => {
   
    if (count > 0) {
      setCount(prev => prev - 1)
    }
  }

  return (
    <div>
      <h2>Count: {count}</h2>

      <button onClick={addCount}>+</button>

      
      {count > 0 && (
        <button onClick={removeCount}> - </button>
      )}
    </div>
  )
}
export default App

import { useState } from 'react'

type Tarefa = {
  id: number;
  atividade: string;
  concluido: boolean;
};

function App () {
  const [valorInput, setValorInput] = useState<string>("")
  const [lista, setLista] = useState<Tarefa[]>([
    {id: 1, atividade: "Tomar café",       concluido: false},
    {id: 2, atividade: "Organizar quarto", concluido: false},
    {id: 3, atividade: "Estudar",          concluido: false},
  ])

  const renderizarLista = () => {
    return lista.map(tarefa => (
      <li key={tarefa.id}>
        <button onClick={() => heandleRemove(tarefa.id)}>Remover</button>
      </li>
    ))
  }


  const heandleRemove = (id: number) => {
    setLista(lista.filter(tarefa => tarefa.id !== id))
  }

  const heandleOnChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    setValorInput(event.target.value)
  }

  const heandleAdd = () => {
    if (valorInput.trim() === "") return
    const novaTarefa: Tarefa = {
      id: Date.now(),
      atividade: valorInput,
      concluido: false,
    }
    setLista([...lista, novaTarefa])
    setValorInput("")
  }

  return (
    <div>
      <h1>Minha Lista de Tarefas</h1>
      <input
        value={valorInput}
        onChange={heandleOnChange}
        placeholder="Digite uma tarefa"
      />
      <button onClick={heandleAdd}>Adicionar</button>
      <ol>{renderizarLista()}</ol>
    </div>
  )
}

export default App
